/* Bubble Sort */
int [10] list;
proc bubbleSort(int [10] list;) {
    int i;
    int j;
    i = 0;
    int temp;
    while (i < 10) do {
        j = 8;
        while (j >= i) do {
            if (list[j+1] > list[j]) then {
                temp = list[j+1];
                list[j+1] = list[j];
                list[j] = temp;
            }
        }
    }
}
call bubbleSort(list,);
