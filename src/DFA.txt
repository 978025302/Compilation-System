# 格式
# 非终结状态 [当前状态 输入符号 下一状态]
# 终结状态 [当前状态 输入符号 下一状态 <种别码,属性值>]
# letter表示a-z、A-Z共52个字母
# number表示0-9共10个数字
# space表示空格
# null表示空串
# start表示初始状态
# stringValue表示属性值为当前终结状态识别的字符串(除空格)
# intNumberValue表示属性值为当前终结状态识别的整型常数(除空格)
# floatNumberValue表示属性值为当前终结状态识别的浮点常数(除空格)
# scienceNumberValue表示属性值为当前终结状态识别的科学常数(除空格)
# numberValue表示属性值为当前终结状态中字符串中的数字(除空格)

# 标识符id
start letter id
start _ id
id letter id
id _ id
id number id
id space ID <ID,stringValue>

# 关键字int
start i i
i n in
in t int
int space INT <INT,->

# 关键字float
start f f
f l fl
fl o flo
flo a floa
floa t float
float space FLOAT <FLOAT,->

# 关键字bool
start b b
b o bo
bo o boo
boo l bool
bool space BOOL <BOOL,->

# 关键字record
start r r
r e re
re c rec
rec o reco
reco r recor
recor d record
record space RECORD <RECORD,->

# 关键字if
start i i
i f if
if space IF <IF,->

# 关键字else
start e e
e l el
el s els
els e else
else space ELSE <ELSE,->

# 关键字do
start d d
d o do
do space DO <DO,->

# 关键字while
start w w
w h wh
wh i whi
whi l whil
whil e while
while space WHILE <WHILE,->

# 关键字declare(过程声明)
start d d
d e de
de c dec
dec l decl
decl a decla
decla r declar
declar e declare
declare space DECLARE <DECLARE,->

# 关键字call(过程调用)
start c c
c a ca
ca l cal
cal l call
call space CALL <CALL,->

# 算术运算符+,-,*,/,++,--
start + PLUS <PLUS,->
start - LESS <LESS,->
start * MULTPLY <MULTPLY,->
start / DIVIDE <DIVIDE,->
PLUS + INC <INC,->
LESS - DEC <DEC,->

# 关系运算符(ASSIGN是赋值符)
start > GT <GT,->
start < LT <LT,->
start = ASSIGN <ASSIGN,->
ASSIGN = EQ <EQ,->
start ! !
! = NE <NE,->
GT = GE <GE,->
LT = LE <LE,->

# 逻辑运算符
start & AND <AND,->
start | OR <OR,->
start ~ NOT <NOT,->

# 界符
start ; END <END,->
start [ [
[ number [?
[? number [?
[? ] ORDER <ORDER,numberValue>
start . POINT <POINT,->

# 常数
start number int_number
int_number number int_number
int_number space INT_NUMBER <INT_NUMBER,intNumberValue>
int_number . float_num
float_num number float_number
float_number number float_number
float_number space FLOAT_NUMBER <FLOAT_NUMBER,floatNumberValue>
INT_NUMBER e e?
FLOAT_NUMBER e e?
e? - e?-
e? number science_number
e-? number science_number
science_number number science_number
science_number space SCIENCE_NUMBER <SCIENCE_NUMBER,scienceNumberValue>
